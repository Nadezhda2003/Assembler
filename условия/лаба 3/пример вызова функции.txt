#include <iostream>
using namespace std;
void   print(int a)
{
	cout << a << '\n';
}
int main()
{
	int n, sum = 0,count = 0,D10=10,r;
	cout << "vvedite chislo" << endl;
	cin >> n;

	_asm
	{
		mov eax, n
		xor ebx, ebx       //кол-во
		m21 : cmp eax, 0
		je m22
		xor edx, edx
		div  D10
		add sum, edx
		mov r, eax
		push edx
		call print
		pop edx
		mov eax,r
		inc  ebx
		jmp  m21
		m22 : mov count, ebx
};
	//sum:=sum+count;
	cout << "sum=" << sum << " count="<<count;
	return 0;
}
